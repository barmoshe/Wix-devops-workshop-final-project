resource "aws_subnet" "barm-terraform-subnet-1" {
  vpc_id             = local.vpc_id
  cidr_block         = var.cidr_1
  availability_zone  = "${var.region}a"
  tags = {
    Name = "barm-terraform-subnet-1"
  }
}

                    

data "aws_vpc" "selected" {
  filter {
    name   = "tag:Name"
    values = ["DevOps-Workshop"]
  }
}

                    

{
  "terraform": {
    "backend": {
      "s3": {
        "bucket": "barm-devops-bucket",
        "key": "terraform.tfstate",
        "region": "eu-west-1"
      }
    }
  },
  "provider": {
    "aws": {
      "region": "eu-west-1"
    }
  }
}

                    

variable "region" {
  type        = string
  default     = "eu-west-1"
  description = "The AWS region in which to create resources."
}

variable "vpc_id" {
  type    = string
  default = "vpc-01b834daa2d67cdaa"
}

variable "use_dynamic_vpc_id" {
  type        = bool
  default     = true
  description = "Set to true if you want to dynamically get the VPC ID by name."
}

variable "cidr_1" {
  type    = string
  default = "192.168.24.0/24"
}

variable "cidr_2" {
  type    = string
  default = "192.168.25.0/24"
}

                    

locals {
  vpc_id = var.use_dynamic_vpc_id ? data.aws_vpc.selected.id : var.vpc_id
}

Summary of Terraform Workflow

  • ✔️ Initialize and configure backend using terraform init.
  • ✔️ Preview changes with terraform plan.
  • ✔️ Apply changes and create resources using terraform apply.
  • ✔️ Manage state with state files stored in S3 (terraform.tfstate).
